hostnamectl set-hostname hq-srv.au-team.irpo
exec bash
timedatectl set-timezone Europe/Moscow
mkdir -p /etc/net/ifaces/eth0
echo "TYPE=eth" > /etc/net/ifaces/eth0/options
echo "BOOTPROTO=static" >> /etc/net/ifaces/eth0/options
echo "192.168.112.2/27" > /etc/net/ifaces/eth0/ipv4address
echo "default via 192.168.112.1" > /etc/net/ifaces/eth0/ipv4route
echo "search au-team.irpo" > /etc/net/ifaces/eth0/resolv.conf
echo "nameserver 127.0.0.1" >> /etc/net/ifaces/eth0/resolv.conf
systemctl restart network
useradd -u 2012 -m -s /bin/bash sshuser
echo "sshuser:P@ssw0rd" | chpasswd
echo "sshuser ALL=(ALL) NOPASSWD: ALL" > /etc/sudoers.d/sshuser
echo "Authorized access only." > /etc/ssh/banner
sed -i 's/#Port 22/Port 2012/' /etc/ssh/sshd_config
sed -i 's/#MaxAuthTries 6/MaxAuthTries 2/' /etc/ssh/sshd_config
echo "AllowUsers sshuser" >> /etc/ssh/sshd_config
echo "Banner /etc/ssh/banner" >> /etc/ssh/sshd_config
systemctl restart sshd
iptables -A INPUT -p tcp --dport 2012 -j ACCEPT
service iptables save
apt-get install -y bind9 bind9utils
cat <<EOF > /etc/bind/named.conf.local
hq-rtr IN A 192.168.112.1
hq-srv IN A 192.168.112.2
hq-cli IN A 192.168.212.10
br-rtr IN A 10.0.0.2
br-srv IN A 192.168.10.2
docker IN A 172.16.30.1
web IN A 172.16.40.1
systemctl restart
